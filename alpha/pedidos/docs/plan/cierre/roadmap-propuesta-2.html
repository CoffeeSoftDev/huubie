Query: Pedidos con saldo pendiente

SELECT
    o.id, o.folio, o.total_pay, o.name,
    COALESCE(SUM(op.pay), 0) AS total_paid,
    (o.total_pay - COALESCE(SUM(op.pay), 0)) AS pending_balance
FROM `order` o
LEFT JOIN order_payments op ON op.order_id = o.id
WHERE DATE(o.date_creation) = :date
AND o.subsidiaries_id = :subsidiary_id
AND o.status != 4 AND o.active = 1
GROUP BY o.id
HAVING pending_balance > 0

Query: Resumen por turno del dia

SELECT
    cs.id, cs.shift_name, cs.opened_at, cs.closed_at,
    cs.total_sales, cs.total_orders,
    cs.total_cash, cs.total_card, cs.total_transfer,
    u.fullname AS opened_by
FROM cash_shift cs
LEFT JOIN fayxzvov_alpha.usr_users u ON u.id = cs.employee_id
WHERE DATE(cs.opened_at) = :date
AND cs.subsidiary_id = :subsidiary_id
AND cs.active = 1
ORDER BY cs.opened_at ASC

Validacion Tipo Accion
